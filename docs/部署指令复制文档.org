* 安装docker docker compose
#+begin_src shell
#!/bin/bash

# 遇到错误立即退出脚本
set -e

echo "=== 开始安装 Docker 和 Docker Compose ==="

echo "1. 更新软件包索引并安装依赖..."
sudo apt-get update
sudo apt-get install -y ca-certificates curl gnupg lsb-release

echo "2. 添加 Docker 的官方 GPG 密钥..."
sudo install -m 0755 -d /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor --yes -o /etc/apt/keyrings/docker.gpg
sudo chmod a+r /etc/apt/keyrings/docker.gpg

echo "3. 设置 Docker 稳定版仓库..."
echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

echo "4. 安装 Docker Engine, CLI, Containerd 和 Docker Compose 插件..."
sudo apt-get update
sudo apt-get install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin

echo "5. 配置当前用户 ($USER) 免 sudo 运行 Docker..."
# 如果 docker 用户组不存在则创建
if ! getent group docker > /dev/null; then
    sudo groupadd docker
fi

# 将当前用户加入 docker 组
sudo usermod -aG docker "$USER"

echo "6. 设置 Docker 开机自启并启动服务..."
sudo systemctl enable docker.service
sudo systemctl enable containerd.service
sudo systemctl start docker.service

echo "================================================================================"
echo "✅ 安装与配置已完成！"
echo ""
echo "⚠️  注意：为了让免 sudo 的用户组权限生效，你需要重新登录你的终端会话。"
echo "你也可以现在直接运行以下命令来立即刷新当前终端的权限："
echo ""
echo "    newgrp docker"
echo ""
echo "验证安装请运行：docker compose version && docker run hello-world"
echo "================================================================================"

#+end_src

    ssl_certificate_key /etc/letsencrypt/live/openclaw.brale.club/privkey.pem;    # SSL 私钥路径


    location / {
        proxy_pass http://127.0.0.1:18789;  # 注意使用 127.0.0.1 而非内网 IP

        # WebSocket 透传（关键！）
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # 必须透传 Origin（配对机制依赖它！）
        proxy_set_header Origin $scheme://$host:$server_port;

        # 其他必要 header
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 超时调大（WebSocket 长连接）
        proxy_read_timeout 86400s;
        proxy_send_timeout 86400s;
    }
}
